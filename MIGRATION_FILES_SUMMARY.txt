COMMON PATTERNS APPLIED TO ALL SERVICES
================================================================================

Pattern 1: Single Record
  OLD: db.query(Model).filter(...).first()
  NEW: result = await db.execute(select(Model).filter(...))
       obj = result.scalar_one_or_none()

Pattern 2: Multiple Records
  OLD: db.query(Model).filter(...).all()
  NEW: result = await db.execute(select(Model).filter(...))
       objs = result.scalars().all()

Pattern 3: Count
  OLD: db.query(Model).count()
  NEW: result = await db.execute(select(func.count(Model.id)))
       total = result.scalar() or 0

Pattern 4: Create/Update
  OLD: db.add(obj); db.commit(); db.refresh(obj)
  NEW: db.add(obj); await db.commit(); await db.refresh(obj)

Pattern 5: Delete
  OLD: db.delete(obj); db.commit()
  NEW: await db.delete(obj); await db.commit()

Pattern 6: Joins
  OLD: db.query(A).join(B).filter(conditions)
  NEW: select(A).join(B).filter(conditions) + await db.execute()

Pattern 7: Aggregates
  OLD: db.query(func.count/avg/sum(...)).scalar()
  NEW: result = await db.execute(select(func.count/avg/sum(...)))
       value = result.scalar() or 0

Pattern 8: Complex Filters
  OLD: query = db.query(...); if ...: query = query.filter(...)
  NEW: query = select(...); if ...: query = query.filter(...)
       result = await db.execute(query)

Pattern 9: Bulk Operations
  OLD: for item in items: db.add(item); db.commit()
  NEW: for item in items: db.add(item); await db.commit()
